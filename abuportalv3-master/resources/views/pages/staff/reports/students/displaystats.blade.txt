    <script>
        $(document).ready(function() {
            // Set up CSRF token for AJAX requests
            $.ajaxSetup({
                headers: {
                    'X-CSRF-TOKEN': $('meta[name="csrf-token"]').attr('content')
                }
            });
            
            // Initialize DataTable
            let dataTable = null;
            
            // Load sessions and levels on page load
            loadSessions();
            loadEntryLevels();
            
            // Faculty change event
            $('#faculty').change(function() {
                const facultyId = $(this).val();
                if (facultyId) {
                    loadDepartments(facultyId);
                    $('#department').prop('disabled', false);
                } else {
                    resetDropdown('#department');
                    resetDropdown('#programmeType');
                    resetDropdown('#programme');
                }
            });
            
            // Department change event
            $('#department').change(function() {
                const departmentId = $(this).val();
                if (departmentId) {
                    loadProgrammeTypes(departmentId);
                    $('#programmeType').prop('disabled', false);
                } else {
                    resetDropdown('#programmeType');
                    resetDropdown('#programme');
                }
            });
            
            // Programme Type change event
            $('#programmeType').change(function() {
                const departmentId = $('#department').val();
                const programmeTypeId = $(this).val();
                if (programmeTypeId) {
                    loadProgrammes(departmentId, programmeTypeId);
                    $('#programme').prop('disabled', false);
                } else {
                    resetDropdown('#programme');
                }
            });
            
            // Form submission
            $('#filterForm').submit(function(e) {
                e.preventDefault();
                fetchStatistics();
            });
            
            // Export button click
            $('#exportButton').click(function() {
                exportToCSV();
            });
            
            // Load departments based on faculty
            function loadDepartments(facultyId) {
                $.ajax({
                    url: "{{ route('get.departments') }}",
                    type: 'GET',
                    data: { faculty_id: facultyId },
                    success: function(data) {
                        populateDropdown('#department', data);
                    },
                    error: function(xhr) {
                        console.error('Error loading departments:', xhr);
                    }
                });
            }
            
            // Load programme types based on department
            function loadProgrammeTypes(departmentId) {
                $.ajax({
                    url: "{{ route('get.programme.types') }}",
                    type: 'GET',
                    data: { department_id: departmentId },
                    success: function(data) {
                        populateDropdown('#programmeType', data, 'prog_type_name');
                    },
                    error: function(xhr) {
                        console.error('Error loading programme types:', xhr);
                    }
                });
            }
            
            // Load programmes based on department and programme type
            function loadProgrammes(departmentId, programmeTypeId) {
                $.ajax({
                    url: "{{ route('get.programmes') }}",
                    type: 'GET',
                    data: { 
                        department_id: departmentId,
                        programme_type_id: programmeTypeId
                    },
                    success: function(data) {
                        populateDropdown('#programme', data);
                    },
                    error: function(xhr) {
                        console.error('Error loading programmes:', xhr);
                    }
                });
            }
            
            // Load entry levels
            function loadEntryLevels() {
                $.ajax({
                    url: "{{ route('get.entry.levels') }}",
                    type: 'GET',
                    success: function(data) {
                        populateDropdown('#level', data);
                    },
                    error: function(xhr) {
                        console.error('Error loading entry levels:', xhr);
                    }
                });
            }
            
            // Load sessions
            function loadSessions() {
                $.ajax({
                    url: "{{ route('get.sessions') }}",
                    type: 'GET',
                    success: function(data) {
                        populateDropdown('#session', data);
                    },
                    error: function(xhr) {
                        console.error('Error loading sessions:', xhr);
                    }
                });
            }
            
            // Fetch statistics based on filters
            function fetchStatistics() {
                $('#loader').show();
                $('#fetchButton span').text('Loading...');
                
                const formData = {
                    faculty_id: $('#faculty').val(),
                    department_id: $('#department').val(),
                    programme_type_id: $('#programmeType').val(),
                    programme_id: $('#programme').val(),
                    session: $('#session').val(),
                    level_id: $('#level').val()
                };
                
                $.ajax({
                    url: "{{ route('reports.students.programmestats') }}",
                    type: 'GET',
                    data: formData,
                    success: function(data) {
                        displayStatistics(data);
                        $('#loader').hide();
                        $('#fetchButton span').text('Fetch Statistics');
                    },
                    error: function(xhr) {
                        console.error('Error fetching statistics:', xhr);
                        $('#loader').hide();
                        $('#fetchButton span').text('Fetch Statistics');
                        alert('Error fetching statistics. Please try again.');
                    }
                });
            }
            
            // Display statistics in the table
            function displayStatistics(data) {
                $('#results').removeClass('d-none');
                
                // Calculate totals
                let totalApplications = 0;
                let totalAdmissions = 0;
                let totalEnrolled = 0;
                
                // Process data for DataTable
                const tableData = data.map(item => {
                    totalApplications += parseInt(item.total_applications);
                    totalAdmissions += parseInt(item.total_admissions);
                    totalEnrolled += parseInt(item.total_enrolled);
                    
                    const admissionRate = item.total_applications > 0 
                        ? ((item.total_admissions / item.total_applications) * 100).toFixed(2) + '%' 
                        : '0%';
                    
                    const enrollmentRate = item.total_admissions > 0 
                        ? ((item.total_enrolled / item.total_admissions) * 100).toFixed(2) + '%' 
                        : '0%';
                    
                    return [
                        item.programme_name,
                        item.department_name,
                        item.prog_type_name,
                        item.total_applications,
                        item.total_admissions,
                        item.total_enrolled,
                        admissionRate,
                        enrollmentRate
                    ];
                });
                
                // Update summary cards
                $('#totalApplications').text(totalApplications);
                $('#totalAdmissions').text(totalAdmissions);
                $('#totalEnrolled').text(totalEnrolled);
                
                // Initialize or refresh DataTable
                if (dataTable) {
                    dataTable.clear().rows.add(tableData).draw();
                } else {
                    dataTable = $('#statisticsTable').DataTable({
                        data: tableData,
                        responsive: true,
                        lengthMenu: [[10, 25, 50, -1], [10, 25, 50, "All"]],
                        order: [[3, 'desc']] // Sort by applications count by default
                    });
                }
            }
            
            // Helper function to populate dropdowns
            function populateDropdown(selector, data, nameField = 'name') {
                const dropdown = $(selector);
                const currentValue = dropdown.val();
                
                // Keep the first option and remove the rest
                dropdown.find('option:not(:first)').remove();
                
                // Add new options
                $.each(data, function(index, item) {
                    dropdown.append($('<option></option>')
                        .attr('value', item.id)
                        .text(item[nameField])
                    );
                });
                
                // If there was a previously selected value, try to reselect it
                if (currentValue) {
                    dropdown.val(currentValue);
                }
                
                // Enable the dropdown
                dropdown.prop('disabled', false);
            }
            
            // Helper function to reset a dropdown
            function resetDropdown(selector) {
                const dropdown = $(selector);
                dropdown.find('option:not(:first)').remove();
                dropdown.val('');
                dropdown.prop('disabled', true);
            }
            
            // Function to export table data to CSV
            function exportToCSV() {
                if (!dataTable) return;
                
                const headers = ['Programme', 'Department', 'Programme Type', 'Applications', 'Admissions', 'Enrollment', 'Admission Rate', 'Enrollment Rate'];
                let csvContent = headers.join(',') + '\n';
                
                // Get data from DataTable
                dataTable.rows().data().each(function(row) {
                    // Quote each field and join with commas
                    const csvRow = row.map(field => `"${field}"`).join(',');
                    csvContent += csvRow + '\n';
                });
                
                // Create CSV file and download
                const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
                const link = document.createElement('a');
                const url = URL.createObjectURL(blob);
                
                const filterDetails = [
                    $('#faculty option:selected').text(),
                    $('#department option:selected').text(),
                    $('#programmeType option:selected').text(),
                    $('#programme option:selected').text(),
                    $('#session option:selected').text(),
                    $('#level option:selected').text()
                ].filter(item => item && !item.includes('Select')).join('_');
                
                const filename = `programme_statistics_${filterDetails || 'all'}_${new Date().toISOString().slice(0,10)}.csv`;
                
                link.href = url;
                link.setAttribute('download', filename);
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
            }
        });
    </script>
