@section("js")
    <script>
        $(function () {
            $('#faculty').on('change', function () {
                // alert()
                $.get('{{route('ajax.department',[':id'])}}'.replace(':id', $(this).val()), function (response) {
                    let options = `<option>Select Department</option>`
                    $.each(response, function (k, v) {
                        options += `<option value='${v.id}'>${v.name}</option>`
                    })
                    $('#department').html(options)
                })
            })

            $('#department, #level').on('change', function() {
                $.get('{{route('staff.ajax.courses',[':deptid',':level'])}}'.replace(':deptid',$('#department').val()).replace(':level',$('#level').val()),
                 function(data) {
                    let options = `<option>Select Course</option>`
                    $.each(data, function(k, v){
                         options += `<option value='${v.id}'>${v.coursecode} - ${v.coursetitle}</option>`
                    })
                    $('#course').html(options)
                })
            
            })

            $(".deleteTeaching").on("click", function (e) {
                e.preventDefault();
                $("#teaching" + ($(this).attr("teaching-id"))).submit()
            });
        })
    </script>
@endsection
