<?php

namespace App\Http\Controllers\Staff;

use App\Models\Employee;
use App\Models\EmployeeContactAddress;
use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;

class EmployeeContactAddressController extends Controller
{
    public function __construct()
    {
        $this->middleware(["auth:staff"]);
    }
    public function index(){
        try{
            $employee = Employee::find(Auth::user()->employee_id);
            return view("pages.staff.profile.contact", compact("employee"));
        }catch (\Exception $e){
            return redirect()->back()->with("error", $e->getMessage());
        }
    }

    public function storeContact(Request $request){
//        return $request;
        try{
            $this->validate($request,[
//                'employee_id' => 'required',
                'address' => 'required',
                'country' => 'required'
            ]);
            $contact = EmployeeContactAddress::find($request->contact_id);
            if(!$contact)
                $contact = new EmployeeContactAddress();
            $contact->employee_id = $request->employee_id;
            $contact->address = $request->address;
            $contact->city = $request->city;
            $contact->state_id = $request->state;
            $contact->country_id = $request->country;
            $contact->postal_code = $request->postal_code;
            $contact->phone = $request->phone;
            $contact->save();
            return redirect()->back()->with("success", 'Done!');

        }catch(\Exception $e){
            return redirect()->back()->with("error", $e->getMessage().' Address and Country are Required');
        }
//        return $request;
    }

    public function deleteContact(EmployeeContactAddress $employeeContactAddress){
//        return $employeeContactAddress;
        try{
            EmployeeContactAddress::destroy($employeeContactAddress->id);
            return redirect()->back()->with('success', 'Done!');
        }catch(\Exception $e){
            return redirect()->back()->with('error', $e->getMessage());
        }
    }
}
