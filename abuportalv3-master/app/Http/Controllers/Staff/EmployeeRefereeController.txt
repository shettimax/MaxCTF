<?php

namespace App\Http\Controllers\Staff;

use App\Models\Employee;
use App\Models\EmployeeReferee;
use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;

class EmployeeRefereeController extends Controller
{
    public function __construct(){
        return $this->middleware(['auth:staff']);
    }

    public function index(){
        try{
            $employee = Employee::find(Auth::user()->employee_id);
            return view('pages.staff.profile.referee',compact('employee'));
        }catch(\Exception $e){
            return redirect()->back()->with('error', $e->getMessage());
        }
    }

    public function storeReferee(Request $request){
        try{
            $this->validate($request, [
                'name' => 'required'
            ]);
            $referee = new EmployeeReferee();
            $referee->employee_id = $request->employee_id;
            $referee->title = $request->title;
            $referee->name = $request->name;
            $referee->position = $request->position;
            $referee->address = $request->address;
            $referee->email = $request->email;
            $referee->phone = $request->phone;
            $referee->save();
            return redirect()->back()->with('success', 'Done!');
        }catch(\Exception $e){
            return redirect()->back()->with('error', $e->getMessage());
        }
    }

    public function deleteReferee(EmployeeReferee $employeeReferee){
        try{
            EmployeeReferee::destroy($employeeReferee->id);
            return redirect()->back()->with('success', 'Done!');
        }catch(\Exception $e){
            return redirect()->back()->with('error', $e->getMessage());
        }
    }
}
