<?php

namespace App\Http\Controllers\Staff;

use App\Models\Employee;
use App\Models\EmployeeWorkExperience;
use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;

class EmployeeWorkExperienceController extends Controller
{
    public function __construct()
    {
        $this->middleware(["auth:staff"]);
    }
    public function index(){
        try{
            $employee = Employee::find(Auth::user()->employee_id);
            return view("pages.staff.profile.workexperience", compact("employee"));
        }catch (\Exception $e){
            return redirect()->back()->with("error", $e->getMessage());
        }
    }

    public function storeWorkExperience(Request $request){
        try{
            $this->validate($request, [
                'place'=>'required',
                'position'=>'required',
                'responsibility'=>'required'
            ]);
            $experience = EmployeeWorkExperience::find($request->experience_id);
            if(!$experience)
                $experience =  new EmployeeWorkExperience();
            $experience->employee_id = $request->employee_id;
            $experience->place = $request->place;
            $experience->position = $request->position;
            $experience->responsibility = $request->responsibility;
            $experience->from = $request->from;
            $experience->to = $request->to;
            $experience->save();
            return redirect()->back()->with('success', 'Done!');
        }catch(\Exception $e){
            return redirect()->back()->with('error', $e->getMessage());
        }
//        return $request;
    }

    public function deleteWorkExperience(EmployeeWorkExperience $employeeWorkExperience){
        //return $employeeWorkExperience;
        try{
            EmployeeWorkExperience::destroy($employeeWorkExperience->id);
            return redirect()->back()->with('success', 'Done!');
        }catch(\Exception $e){
            return redirect()->back()->with('error', $e->getMessage());
        }
    }
}
