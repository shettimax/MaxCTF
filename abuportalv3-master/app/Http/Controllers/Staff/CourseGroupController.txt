    public function getCourses()
    {
        /**$rules = ["faculty"=>"required|integer","department"=>"required|integer","session"=>"required|integer", "semester"=>"required|integer"];
         * $messages = ['faculty.integer' => 'Faculty must be provided', 'department.integer' => 'Department must be provided', 'session.required' => 'Session must be provided', 'semester.required' => 'Semester must be provided' ];
         * $this->validate(request(), $rules, $messages);*/
        try {
            $session = request('session');
            $semester = request('semester');
            $courses = Course::where(['coursesemester' => request('semester'), 'department_id' => request('department')])
                ->leftJoin('course_groups', function ($join) use ($session) {
                    $join->on('courses.id', '=', 'course_groups.course_id')
                        ->on('session', "=", DB::raw($session));
                })
                ->orderBy('courses.coursecode')
                ->groupBy('courses.id')
                ->select(['coursecode', 'coursetitle', DB::raw("IF(session,session,'$session') AS session"), DB::raw("IF(semester,semester,'$semester') AS semester"), DB::raw("IF(course_id,course_id,courses.id) AS course_id")])
                ->get();
        } catch (\Exception $e) {
            return $e->getMessage();

        }
        return $courses;
    }

    public function list($session, $semester, $course_id)
    {
        $groups = CourseGroup::where(['session' => $session, 'semester' => $semester, 'course_id' => $course_id])->get();

//        return $groups;
        $course = Course::find($course_id);
        if ($groups->isNotEmpty()) {
            $otherdetails = ['session' => $groups->first()->session,
                'semester' => $groups->first()->semester, 'coursecode' => $groups->first()->course->coursecode,
                'coursetitle' => $groups->first()->course->coursetitle];
        } else {
            $otherdetails = ['session' => $session,
                'semester' => $semester, 'coursecode' => $course->coursecode,
                'coursetitle' => $course->coursetitle];
        }

        $lecturegroups = collect([]);
        $labgroups = collect([]);
        foreach ($groups as $group) {
            if ($group->grouptype == "LT") {
                $lecturegroups->push($group);
            } else {
                $labgroups->push($group);
            }
        }

        return view('pages.staff.coursegroup.list', compact('lecturegroups', 'labgroups',
            'otherdetails', 'session', 'semester', 'course_id'));
//        } else {
//            return view('coursegroup.empty');
//        }

    }

    public function createGroup($session, $semester, $course_id, $grouptype, $maxsize)
    {

        $coursegroup = CourseGroup::where(['session' => $session, 'semester' => $semester, 'course_id' => $course_id, 'grouptype' => $grouptype])->orderBy('groupno', 'desc')->first();

        $groupno = (!$coursegroup) ? 1 : $coursegroup->groupno + 1;


        //$groupno = $coursegroup->groupno + 1;
        $coursegroup = new CourseGroup();
        $coursegroup->groupno = $groupno;
        $coursegroup->course_id = $course_id;
        $coursegroup->grouptype = $grouptype;
        $coursegroup->session = $session;
        $coursegroup->semester = $semester;
        $coursegroup->maxsize = $maxsize;
        $coursegroup->save();
        return array($coursegroup->id);
    }

    public function savePeriod($courseperiod, $course_group_id, $day, $starttime, $endtime, $venue)
    {
        $courseperiod->course_group_id = $course_group_id;
        $courseperiod->day = $day;
        $courseperiod->start = $starttime;
        $courseperiod->end = $endtime;
        $courseperiod->facility_id = $venue;
        $courseperiod->save();
    }


    public function save($session, $semester, $course_id)
    {
//        return [$session,$semester,$course_id];
        try {
            if (!$this->access(Auth::user(), "coursegroup.index"))
                return view("pages.errors.nopermission");

            $session = $session;
            $semester = $semester;
            $course_id = $course_id;
            $grouptype = request('grouptype');
            $period = request('period_' . $grouptype);
            $groupno = request('groupno_' . $grouptype);
            $groupid = request('groupid_' . $grouptype);
            $day = request('day_' . $grouptype);
            $starttime = request('starttime_' . $grouptype);
            $endtime = request('endtime_' . $grouptype);
            $venue = request('venue_' . $grouptype);
            $maxsize = request('maxsize_' . $grouptype);


            /** Add or update course period based on course group, otherwise add new course group and course period*/
            if ($groupno != 'New') {
                $courseperiod = CoursePeriod::find($period);
                if (!$courseperiod) {
                    $courseperiod = new CoursePeriod();
                }
                $this->savePeriod($courseperiod, $groupid, $day, Carbon::parse($starttime)->toTimeString(), Carbon::parse($endtime)->toTimeString(), $venue);
            } else {
                $coursegroup = $this->createGroup($session, $semester, $course_id, $grouptype, $maxsize);
                $course_group_id = $coursegroup[0];
                $this->savePeriod(new CoursePeriod(), $course_group_id, $day, Carbon::parse($starttime)->toTimeString(), Carbon::parse($endtime)->toTimeString(), $venue);
            }

            return redirect()->route('coursegroup.list', ['session' => $session, 'semester' => $semester, 'course' => $course_id]);
        } catch (\Exception $e) {
            return $e->getMessage();
//            return view("pages.staff.error.technical");
        }

    }

    public function addemployee($session, $semester, $course_id)
    {
        $employeeid = request('employee');
        $groupid = request('groupid');
        if (CourseLecturer::where(['course_group_id' => $groupid, 'employee_id' => $employeeid])->get()->isEmpty()) {
            $courselecturer = new CourseLecturer();
            $courselecturer->course_group_id = $groupid;
            $courselecturer->employee_id = $employeeid;
            $courselecturer->save();
        }
        return redirect()->route('coursegroup.list', ['session' => $session, 'semester' => $semester, 'course' => $course_id]);
    }

    public function deleteGroup()
    {
        try {
            if (!$this->access(Auth::user(), 'coursegroup.index')) {
                return view("pages.staff.error.access");
            }
            $groupid = request('groupid');
            $coursereg = CourseRegistration::where(['course_group_id' => $groupid])->first();
            if ($coursereg)
                //Cannot be deleted due to entry in course registration
                return -1;
            else {
                //Delete records that matches groupid in coursegroups and course periods
                CoursePeriod::where(['course_group_id' => $groupid])->delete();
                CourseGroup::where(['id' => $groupid])->delete();
                return 1;
            }
        } catch (\Exception $e) {
            return view("pages.staff.error.technical");
        }

    }

    public function loadCourses(Request $request)
    {
        return $request;
        $matricnumber = htmlspecialchars($request->input('matricnumber'));
        $session = htmlspecialchars($request->input('session'));

        try {
            // Query to fetch course details
            $courses = DB::table('course_registrations')
                ->join('course_groups', 'course_registrations.course_group_id', '=', 'course_groups.id')
                ->join('students', 'students.id', '=', 'course_registrations.student_id')
                ->join('courses', 'courses.id', '=', 'course_groups.course_id')
                ->join('programmes', 'programmes.id', '=', 'students.programme_id')
                ->join('departments', 'programmes.department_id', '=', 'departments.id')
                ->join('faculties', 'departments.faculty_id', '=', 'faculties.id')
                ->select('students.matric_number', 'firstname', 'surname', 'other_names', 'programmes.name as Programme',
                    'coursecode', 'coursetitle', 'courseunit', 'semester', 'groupno', 'grouptype')
                ->where('course_groups.session', $session)
                ->where('students.matric_number', 'LIKE', $matricnumber)
                ->orderBy('semester')
                ->orderBy('coursecode')
                ->get();

            // Query to fetch student details
            $studentDetails = DB::table('students')
                ->join('programmes', 'programmes.id', '=', 'students.programme_id')
                ->join('departments', 'programmes.department_id', '=', 'departments.id')
                ->join('faculties', 'departments.faculty_id', '=', 'faculties.id')
                ->select('students.matric_number', 'firstname', 'surname', 'other_names', 'programmes.name as Programme')
                ->where('matric_number', $matricnumber)
                ->distinct()
                ->first();

            if ($studentDetails) {
                echo "<p><b>Matric Number: </b>" . htmlspecialchars($studentDetails->matric_number) . "</p>";
                echo "<p><b>Full Name: </b>" . htmlspecialchars($studentDetails->firstname) . " " . htmlspecialchars($studentDetails->surname) . " " . htmlspecialchars($studentDetails->other_names) . "</p>";
                echo "<p><b>Programme:</b> " . htmlspecialchars($studentDetails->Programme) . "</p>";
            }

            $courseArray = [];
            foreach ($courses as $course) {
                $coursecode = $course->coursecode;
                $grouptype = $course->grouptype;

                if (!isset($courseArray[$coursecode])) {
                    $courseArray[$coursecode] = (array) $course;
                    $courseArray[$coursecode]['LTGroup'] = '';
                    $courseArray[$coursecode]['LBGroup'] = '';
                }

                if ($grouptype === 'LT') {
                    $courseArray[$coursecode]['LTGroup'] = $course->groupno;
                } elseif ($grouptype === 'LB') {
                    $courseArray[$coursecode]['LBGroup'] = $course->groupno;
                }
            }

            $sum_sem1 = 0;
            $sum_sem2 = 0;
            $count = 1;
            $prev_semester = 0;

            echo '<form id="deleteCoursesForm">';
            echo '<table>
                    <tr style="color:blue;">
                        <th>SN</th>
                        <th>Coursecode</th>
                        <th>Coursetitle</th>
                        <th>Semester</th>
                        <th>Courseunit</th>
                        <th>LT Group</th>
                        <th>LB Group</th>
                        <th>Drop</th>
                    </tr>';

            foreach ($courseArray as $course) {
                if ($course['semester'] == 1) {
                    $sum_sem1 += $course['courseunit'];
                } elseif ($course['semester'] == 2) {
                    $sum_sem2 += $course['courseunit'];
                }

                if ($prev_semester == 1 && $course['semester'] == 2) {
                    echo "<tr><td colspan='8'>&nbsp;</td></tr>";
                }

                echo "<tr>
                        <td>" . $count . "</td>
                        <td>" . htmlspecialchars($course['coursecode']) . "</td>
                        <td>" . htmlspecialchars($course['coursetitle']) . "</td>
                        <td>" . htmlspecialchars($course['semester']) . "</td>
                        <td>" . htmlspecialchars($course['courseunit']) . "</td>
                        <td>" . htmlspecialchars($course['LTGroup']) . "</td>
                        <td>" . htmlspecialchars($course['LBGroup']) . "</td>
                        <td><input type='checkbox' name='coursesdelete[]' value='" . htmlspecialchars($course['coursecode']) . "'></td>
                      </tr>";

                $count++;
                $prev_semester = $course['semester'];
            }

            echo '</table>';
            echo "<p>Total CU 1st Semester: " . $sum_sem1 . "</p>";
            echo "Total CU 2nd Semester: " . $sum_sem2;
            echo '<div class="center">
                    <button type="submit" style="padding: 4px 7px; margin-left:80%;color:red; font-size: 12px; cursor: pointer;">Drop Selected Courses</button>
                  </div>';
            echo '</form>';

        } catch (\Exception $e) {
            return response()->json(['error' => 'Connection failed: ' . $e->getMessage()], 500);
        }
    }
