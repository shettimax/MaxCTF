    public function resetPassword(Request $request)
    {
        try {
            $usr = substr(decrypt($request->uilwaknscgfdjp), 0, -7);

            $user = User::where(['username' => $request->$usr])
                ->first();

            if ($user) {
                // Generate a 6-digit alphanumeric code
                $resetCode = strtoupper(Str::random(8));

                //  Change the staff password
                $staff = $user->employee();
                $user->password = bcrypt($resetCode);
                $user->save();

                Mail::to([$user->email, $staff->email])->send(new PasswordResetMail(
                    $staff,
                    $resetCode
                ));

                return redirect('staff')->with(["success" => "Reset code sent to your email"]);
            }

            return redirect()->back()->withErrors(["message" => "Wrong username or email address"]);
        } catch (\Exception $e) {
            return redirect()->back()->withErrors(["message" => "Looks like something went wrong: " . $e->getMessage()]);
        }
    }

    public function logout(Request $request)
    {
        Auth::guard("staff")->logout();
        $request->session()->invalidate();
        return redirect(route('staff'));
