<?php

namespace App\Notifications;

use Illuminate\Bus\Queueable;
use Illuminate\Notifications\Notification;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Notifications\Messages\MailMessage;

class ReservationExpiredNotification extends Notification implements ShouldQueue
{
    use Queueable;

    protected $reservation;

    public function __construct($reservation)
    {
        $this->reservation = $reservation;
    }

    public function via($notifiable)
    {
        return ['mail'];
    }

    public function toMail($notifiable)
    {
        return (new MailMessage)
            ->subject('Your Bedspace Reservation Has Expired')
            ->greeting("Dear {$notifiable->name},")
            ->line("Your bedspace reservation for session {$this->reservation->session} has expired because payment was not completed within 48 hours.")
            ->line("The space has now been made available to others.")
            ->action('Log in to the Student Portal to re-apply', url('portal.abu.edu.ng'))
            ->line('If you have any issues, please contact the Student Affairs Office.');
    }
}
