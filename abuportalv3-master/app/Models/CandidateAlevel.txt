<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;

class CandidateAlevel extends Model
{
    protected $guarded = [];
    public static $rules = [
        'inst' => 'required|min:5',
        'fromyear' => 'required|min:4|max:4',
        'toyear' => 'required|min:4|max:4',
        'qual' => 'required|min:1',
        'degree' => 'required|min:1',
        'course' => 'required|min:6',
        'cgpa' => 'required|numeric'
    ];

    public function qualification()
    {
        return $this->belongsTo(Qualification::class);
    }

    public function alevelGrades()
    {
        return $this->belongsTo(QualificationGrade::class,'degree_class_id');
    }

    public function qualName()
    {
        $qualification = $this->qualification()->first();
        return $qualification?$qualification->qualification:"";
    }

    public function grade()
    {
        $grade = $this->alevelGrades()->first();
        return $grade?$grade->grade:"";
    }
}
