<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;

class AccommodationSessionFee extends Model
{
    use HasFactory;

    protected $fillable = ['student_type', 'unit_type', 'unit_id', 'session', 'amount'];

    public function programmeType()
    {
        return $this->hasOne(ProgrammeType::class, 'prog_type_code', 'student_type');
    }
}
