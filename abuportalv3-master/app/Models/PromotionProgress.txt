<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;

class PromotionProgress extends Model
{
    use HasFactory;

    protected $fillable = [
        'promotion_exercise_id',
        'promotion_recomendation_id',
        'step_id',
        'step_configuration_id',
        'status',
        'approver_id',
        'approver_comment',
        'approved_at'
    ];

    protected $casts = [
        'approved_at' => 'datetime',
    ];

    public function promotionExercise()
    {
        return $this->belongsTo(PromotionExercise::class);
    }

    public function promotionRecommendation()
    {
        return $this->belongsTo(PromotionRecomendation::class);
    }

    public function step()
    {
        return $this->belongsTo(PromotionStep::class);
    }

    public function stepConfiguration()
    {
        return $this->belongsTo(PromotionStepConfiguration::class);
    }

    public function approver()
    {
        return $this->belongsTo(User::class, 'approver_id');
    }

    public function scopePending($query)
    {
        return $query->where('status', 'Pending');
    }

    public function scopeCompleted($query)
    {
        return $query->where('status', 'Completed');
    }

    public function scopeRejected($query)
    {
        return $query->where('status', 'Rejected');
    }

    public function scopeInProgress($query)
    {
        return $query->where('status', 'In Progress');
    }

    public function isCompleted()
    {
        return $this->status === 'Completed';
    }

    public function isPending()
    {
        return $this->status === 'Pending';
    }

    public function isRejected()
    {
        return $this->status === 'Rejected';
    }
}
