<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Model;

class PaymentForWhat extends Model
{
    //
    public static $rules = [
        "itemname"=>"required|exists:fee_items,id",
        "residency"=>"required",
        "newreturning"=>"required",
        "faculty"=>"required",
        "department"=>"required",
        "programme"=>"required",
        "level"=>"required",
        "artsscience"=>"required",
        "progtype"=>"required",
        "progmode"=>"required",
        "amount"=>"required|integer",
        "staff"=>"required",
        "effective"=>"required",
        "religion"=>"required"
    ];
    public function feesettings(){
        return $this->hasMany('App\Models\FeeSettings','payment_for_what_id');
    }
}
