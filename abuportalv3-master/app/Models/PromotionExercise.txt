<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;
use Carbon\Carbon;

class PromotionExercise extends Model
{
    use HasFactory;

    protected $fillable = [
        'year',
        'title',
        'description',
        'start_date',
        'end_date',
        'status'
    ];

    protected $casts = [
        'start_date' => 'date',
        'end_date' => 'date',
    ];

    public function stepConfigurations()
    {
        return $this->hasMany(PromotionStepConfiguration::class);
    }

    public function progress()
    {
        return $this->hasMany(PromotionProgress::class);
    }

    public function isActive()
    {
        return $this->status === 'Active' &&
               Carbon::now()->between($this->start_date, $this->end_date);
    }

    public function isOpen()
    {
        return Carbon::now()->lt($this->end_date);
    }

    public function scopeActive($query)
    {
        return $query->where('status', 'Active');
    }

    public function scopeCurrent($query)
    {
        return $query->where('status', 'Active')
                    ->where('start_date', '<=', Carbon::now())
                    ->where('end_date', '>=', Carbon::now());
    }
}
